int live_life_on_the_edge() {
    int lo = 0, hi = 1e9; 
    while (lo < hi) {
        mid = (lo + hi + 1)/2; //mid = courage score
        if (alive(mid)) {
            lo = mid;
        }
        else { //dead
            hi = mid-1;
        }
    }
    return lo;
}
